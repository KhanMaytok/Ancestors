# ACS.00042 - Ensure only one Vee
character_event = {
	id = ACS.00042
	is_triggered_only = yes
	hide_window = yes
	ai = no

    trigger = {
		OR = {
			multiplayer = no
			AND = {
				multiplayer = yes
				is_multiplayer_host_character = yes
			}
		}
	}

	immediate = {
		random_character = {
            limit = {            
                has_character_flag = is_vee
            }
            save_event_target_as = dont_kill
        }
        any_character = {
            limit = {
                has_character_flag = is_vee
                NOT = { character = event_target:dont_kill }
            }
            death = { death_reason = death_platypus }
        }
	}
}

# ACS.00044 - Spread ancestor chests
character_event = {
	id = ACS.00044
	is_triggered_only = yes
	hide_window = yes
	ai = no

    trigger = {
		NOT = { has_global_flag = spawned_ancestors_chests }
		OR = {
			multiplayer = no
			AND = {
				multiplayer = yes
				is_multiplayer_host_character = yes
			}
		}
	}

	immediate = {
		any_province = {
            limit = {            
                is_land = yes
            }
            add_province_modifier = { name = ancestor_chest duration = -1 }			
        }
		set_global_flag = spawned_ancestors_chests
	}
}
